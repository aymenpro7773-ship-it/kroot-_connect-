📱 تطبيق أتمتة الرد على رسائل SMS

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✨ الميزات الرئيسية:

✅ أتمتة الرد على رسائل SMS من مرسل محدد (jaib)
✅ تحليل ذكي للرسالة لاستخراج المبلغ ورقم العميل
✅ إرسال تلقائي لكود الكرت المناسب
✅ إدارة الكروت بفئات (100، 200، 300، 500 ر.ي)
✅ لوحة تحكم متقدمة مع إحصائيات
✅ نظام سجلات شامل
✅ إشعارات ذكية وتنبيهات
✅ أمان عالي للبيانات

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🎨 التصميم:

• واجهة عصرية وملونة
• ألوان زاهية وجذابة
• تدرجات لونية جميلة
• بطاقات مستديرة وظلال ناعمة
• سهلة الاستخدام

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 المتطلبات:

• Android 5.0 (API 21) أو أحدث
• 50 MB مساحة تخزين
• الأذونات:
  - استقبال الرسائل (RECEIVE_SMS)
  - إرسال الرسائل (SEND_SMS)
  - قراءة الرسائل (READ_SMS)
  - الوصول للجهات (READ_CONTACTS)
  - الإشعارات (POST_NOTIFICATIONS)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🚀 كيفية التثبيت:

1. انسخ ملف APK إلى هاتفك
2. افتح الملف
3. اضغط "تثبيت"
4. انتظر انتهاء التثبيت
5. افتح التطبيق

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

⚙️ الإعدادات الأولية:

1. افتح الإعدادات
2. أدخل اسم المرسل: jaib
3. حدد حد التنبيه: 5
4. فعّل الإشعارات
5. احفظ الإعدادات

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

• ملفات Java: 15
• ملفات XML: 12+
• أسطر الكود: ~3,500+
• حجم التطبيق: ~50 MB
• الإصدار: 1.0
